    simu = cathy_tools.CATHY(dirName=path2prj, 
    			prj_name="weil_exemple_outputs_plot"
    			)

    simu.run_preprocessor()
    simu.run_processor(IPRT1=2, 
                        DTMIN=1e-2,
                        DTMAX=1e2,
                        DELTAT=5,
                       TRAFLAG=0,
                       verbose=False
                       )
